// server.js
const express = require('express');
const fetch = require('node-fetch');
const app = express();
const PORT = process.env.PORT || 3000;

// Simple proxy: GET /proxy?url=https://example.com
app.get('/proxy', async (req, res) => {
  const url = req.query.url;
  if (!url) return res.status(400).send('Missing url parameter');
  try {
    const r = await fetch(url, { method: 'GET', headers: { 'User-Agent': 'ViewSource-Proxy/1.0' } });
    const text = await r.text();
    res.set('Content-Type', 'text/html; charset=utf-8');
    res.send(text);
  } catch (err) {
    res.status(500).send('Proxy error: ' + String(err));
  }
});

app.listen(PORT, ()=> console.log('Proxy listening on port', PORT));
Run: node server.js. Then check http://localhost:3000/proxy?url=... in the tool by enabling "Use local proxy".
